function compute_disorder(stack a):
    mistakes = 0
    total_pairs = 0
    for i from 0 to size(a)-1:
        for j from i+1 to size(a)-1:
        total_pairs += 1

        if a[i] > a[j]:
            mistakes += 1
    return mistakes / total_pairs
